1033 
1034 		// built-ins on libraries
1035 		Set<PrologIndicator> builtins = new HashSet<PrologIndicator>();
1036 		Enumeration<?> e = ((JLogEngine) engine).engine.getPredicateRegistry().enumPredicates();
1037 		while (e.hasMoreElements()) {
1038 			Object object = e.nextElement();
1039 			if (object instanceof pGenericPredicateEntry) {
1040 				pGenericPredicateEntry entry = (pGenericPredicateEntry) object;
1041 				String functor = entry.getName();
1042 				int arity = entry.getArity();
1043 				JLogIndicator pi = new JLogIndicator(functor, arity);
1044 				builtins.add(pi);
1045 			}
1046 		}
1047 
1048 		// user defined predicates
1049 		e = ((JLogEngine) engine).engine.getKnowledgeBase().enumDefinitions();
1050 		while (e.hasMoreElements()) {
1051 			jRuleDefinitions definitions = (jRuleDefinitions) e.nextElement();
1052 			Enumeration<?> rules = definitions.enumRules();
1053 			while (rules.hasMoreElements()) {
1054 				Object object2 = rules.nextElement();
1055 				if (!(object2 instanceof jBuiltinRule)) {
1056 					jRule jRule = (jRule) object2;
1057 					jPredicate ruleHead = jRule.getHead();
1058 					String functor = ruleHead.getName();
1059 					int arity = ruleHead.getArity();
1060 					JLogIndicator pi = new JLogIndicator(functor, arity);
1061 					builtins.add(pi);
1062 				}
1063 			}
1064 		}
1065 
1066 		assertEquals(builtins, engine.currentPredicates());
1067 	}
