package io.github.prolobjectlink.prolog.jlog;

import static io.github.prolobjectlink.prolog.PrologTermType.DOUBLE_TYPE;

import io.github.prolobjectlink.prolog.PrologDouble;
import io.github.prolobjectlink.prolog.PrologProvider;

/**
 * 
 * @author Jose Zalacain
 * @since 1.0
 */
final class JLogDouble extends JLogFloat implements PrologDouble {

	JLogDouble(PrologProvider provider, Number value) {
		super(DOUBLE_TYPE, provider, new jDouble(value.doubleValue()));
	}

	public final double getDoubleValue() {
		return ((jDouble) value).getRealValue();
	}

	public float getFloatValue() {
		return (float) getDoubleValue();
	}

}
